  id: number;
  nombre: string;
  activo: boolean;
}

function mostrar(
  u: Usuario
): string {
  return `[${u.id}] ${u.nombre}`;
}

const jose: Usuario = {

SQL / MySQL

Base de Datos Relacional

1974

El estándar para gestionar bases de datos. Permite crear, consultar, actualizar y eliminar información. MySQL es su versión open source más popular, usada en la mayoría de aplicaciones web.

ConsultasTablasMySQLRelaciones

CREATE TABLE alumnos (
  id     INT AUTO_INCREMENT,
  nombre VARCHAR(100),
  grado  INT,
  PRIMARY KEY(id)
);

-- Insertar y consultar
INSERT INTO alumnos
  VALUES (NULL,'Jose',3);

SELECT * FROM alumnos
WHERE grado = 3;

Sass / SCSS

Preprocesador CSS

2006

El preprocesador CSS más popular. Permite variables, anidación, mixins, funciones y herencia para estilos más potentes. El código SCSS se compila a CSS estándar compatible con todos los navegadores.

VariablesMixinsNestingCompilado
estilos.scss 
$azul: #2563eb;
$radio: 12px;

@mixin sombra($color) {
  box-shadow: 0 4px 20px $color;
}

.boton {
  background: $azul;
  border-radius: $radio;
  @include sombra(
    rgba($azul, .4)
  );
  &:hover {
    background: darken($azul,10%);
  }
}
